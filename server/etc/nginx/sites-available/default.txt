    listen 443 ssl;
    server_name eth-node-map.xyz;

    ssl_certificate /etc/letsencrypt/live/eth-node-map.xyz/fullchain.pem;

    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    index index.html;       

    location / {
        try_files $uri $uri/ =404;
